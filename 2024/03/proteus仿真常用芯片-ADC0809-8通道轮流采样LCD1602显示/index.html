51单片机 ADC0809模数转换与显示+Proteus仿真

📑本示例将ADC08098个通道的采样数据分别显示在LCD1602对应位置，保留小数点后2位精度。这样依赖lcd1602整个屏幕刚好显示8组数据，中间没有间隔空余。

📓功能说明

  • 🌿定时器0用来给ADC0809提供时钟源。

  • 🌿定时器1用来产生0.5s定时，用来查询各个ADC通道数据。

  • 🔰如果想单通道显示可以启用外部中断0，用来作为通道切换的按键。

  • ✨如果觉得不美观，当然也可以只显示单通道，然后使用按键来切换通道显示,，调整显示函数。

void exint0() interrupt 0 //(location at 0003H) { n++; if(n==8)n=0; } //在main中添加：，用来开启外部中断 IT0 = 1; //set INT0 int type (1:Falling 0:Low level)下降沿触发 EX0 = 1; //enable INT0 interrupt

  • 工程架构
  • 相关未启用的函数警告处理：移除未使用的函数。

📖main主程序

`#include “LCD1602.h”

sbit key1= P1^6;
sbit key2= P1^7;
unsigned char dat,results;
volatile unsigned char n,CNT;

sbit CLK=P3^5;
/ADC通道选择/
sbit P10=P1^0;
sbit P11=P1^1;
sbit P12=P1^2;

sbit EOC = P3^1;
sbit OE = P3^0;
sbit START = P3^4 ;

void Timer0Init(void) //100微秒@12.000MHz
{
TMOD &= 0xF0; //设置定时器模式
TMOD |= 0x02; //设置定时器模式
TL0 = 0x9C; //设置定时初始值
TH0 = 0x9C; //设置定时重载值
TF0 = 0; //清除TF0标志
TR0 = 1; //定时器0开始计时
}

//定时器中断
void Timer0_INT() interrupt 1
{
CLK =!CLK;//给ADC0809提供时钟频率
TF0 = 0; //清除TF0标志
TL0 = 0x9C; //设置定时初始值100微秒=10KHz
TH0 = 0x9C; //设置定时重载值

}
/*
//External interrupt0 service routine外部中断0：P32
void exint0() interrupt 0 //(location at 0003H)
{
n++;
if(n==8)n=0;
}
*/

void Timer1Init(void) //50毫秒@12.000MHz
{
TMOD &= 0x0F; //设置定时器模式
TMOD |= 0x10; //设置定时器模式
TL1 = 0xB0; //设置定时初始值
TH1 = 0x3C; //设置定时初始值
TF1 = 0; //清除TF0标志
TR1 = 1; //定时器0开始计时
}

/* Timer1 interrupt routine */
void tm1_isr() interrupt 3
{
TF1 = 0; //清除TF0标志
TL0 = 0xB0; //设置定时初始值
TH0 = 0x3C; //设置定时初始值

if (CNT++ == 10)   //50ms * 20 -> 2s
{
    CNT = 0;  //归零
    n++;
            if(n==8)n=0;
}

}
void Channel(unsigned char x)
{

switch(x){
case 0: P10=0; P11=0; P12=0;break;
case 1: P10=1; P11=0; P12=0;break;
case 2: P10=0; P11=1; P12=0;break;
case 3: P10=1; P11=1; P12=0;break;
case 4: P10=0; P11=0; P12=1;break;
case 5: P10=1; P11=0; P12=1;break;
case 6: P10=0; P11=1; P12=1;break;
case 7: P10=1; P11=1; P12=1;break;
}

}

void main(void)
{
unsigned char Int,Dec;
unsigned int Value;
// IT0 = 1; //set INT0 int type (1:Falling 0:Low level)
// EX0 = 1; //enable INT0 interrupt
Timer0Init();
Timer1Init();
LCMInit(); //将液晶初始化
ET0 =1;//开启定时器0
ET1 = 1;//开启定时器1
EA =1;//开启总中断
DisplayListChar(0,0,5, “ . . . .”);//字符串显示
DisplayListChar(1,0,5, “ . . . .”);//字符串显示
while(1)
{
delayms(16);
Channel(n); //选择通道

  START=0;     
  START=1;
  START=0;//当低电平来时开始读取数据
  while(EOC==0);
      OE=1;
     dat = P2; //将数据赋给P2
      OE = 0;       
 Value = (dat*100)/51;
 Int = dat/51;//dat*5/255
 Dec = Value%100;//处理数据
if(n<4){
  DisplayOneChar(0 ,4*n,Int+0x30);
    DisplayOneChar(0 ,2+4*n,Dec/10 +0x30);
DisplayOneChar(0 ,3+4*n,Dec%10 +0x30);
}		
else{
    DisplayOneChar(1 ,4*n,Int+0x30);
    DisplayOneChar(1 ,2+4*n,Dec/10 +0x30);
DisplayOneChar(1 ,3+4*n,Dec%10 +0x30);
}

}
}`
